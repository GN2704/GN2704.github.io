    <script>
        document.addEventListener('DOMContentLoaded', () => {
            const linkItems = document.querySelectorAll('dl dt');

            linkItems.forEach((item, index) => {
                const checkbox = document.createElement('input');
                checkbox.type = 'checkbox';

                // **1. Čitanje sačuvane vrednosti iz LocalStorage-a**
                const isChecked = localStorage.getItem('checkbox_' + index) === 'true';
                checkbox.checked = isChecked;

                // **2. Dodavanje događaja za čuvanje stanja**
                checkbox.addEventListener('change', () => {
                    if (checkbox.checked) {
                        localStorage.setItem('checkbox_' + index, 'true');
                    } else {
                        localStorage.removeItem('checkbox_' + index);
                    }
                });
                
                item.prepend(checkbox);
            });
        });
    </script>
