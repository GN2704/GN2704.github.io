            // Try to restore from storage, if fails use original content
            if (!restoreFromStorage()) {

        function initializeDataIds() {
            // Add data-id to elements that don't have it
            let idCounter = nextElementId;

            // Sections
            document.querySelectorAll('.main-section').forEach((section, index) => {
                if (!section.dataset.id) {
                    section.dataset.id = `section_${index + 1}`;
                }
            });

            // Folders
            document.querySelectorAll('.folder').forEach((folder) => {
                if (!folder.dataset.id) {
                    folder.dataset.id = `folder_${idCounter++}`;
                }
            });

            nextElementId = idCounter;
        }

        function initializeOriginalFunctionality() {
            // Search bar drag functionality
            const dragElement = document.getElementById("searchBar");
            let offsetX = 0,
                offsetY = 0,
                isDown = false;

            if (dragElement) {
                dragElement.addEventListener('mousedown', function(e) {
                    isDown = true;
                    offsetX = e.clientX - dragElement.offsetLeft;
                    offsetY = e.clientY - dragElement.offsetTop;
                    dragElement.style.cursor = 'grabbing';
                });

                document.addEventListener('mouseup', function() {
                    isDown = false;
                    if (dragElement) dragElement.style.cursor = 'grab';
                });

                document.addEventListener('mousemove', function(e) {
                    if (!isDown || !dragElement) return;
                    e.preventDefault(); // Prevent text selection while dragging
                    dragElement.style.left = (e.clientX - offsetX) + 'px';
                    dragElement.style.top = (e.clientY - offsetY) + 'px';
                });
            }

            // Folder click functionality
            attachFolderClickHandlers();

            // Toggle Sections button
            const toggleSectionsButton = document.getElementById('toggleSections');
            if (toggleSectionsButton) {
                toggleSectionsButton.addEventListener('click', function() {
                    const allSectionContents = document.querySelectorAll('.section-content');
                    const isAnyClosed = Array.from(allSectionContents).some(c => !c.classList.contains('open'));

                    allSectionContents.forEach((content) => {
                        if (isAnyClosed) {
                            content.classList.add('open');
                        } else {
                            content.classList.remove('open');
                        }
                    });

                    toggleSectionsButton.innerHTML = isAnyClosed ?
                        '<i class="fas fa-compress-arrows-alt"></i> Zatvori sve sekcije' :
                        '<i class="fas fa-expand-arrows-alt"></i> Otvori sve sekcije';
                });
            }

            // Toggle All Folders button
            const toggleAllBtn = document.getElementById('toggleAll');
            if (toggleAllBtn) {
                toggleAllBtn.addEventListener('click', function() {
                    const allContents = document.querySelectorAll('.folder-content');
                    
                    const allIcons = document.querySelectorAll('.folder-header .folder-icon');

                    const isAnyClosed = Array.from(allContents).some(c => !c.classList.contains('open'));

                    allContents.forEach(content => {
                        if (isAnyClosed) {
                            content.classList.add('open');
                        } else {
                            content.classList.remove('open');
                        }
                    });

                    allIcons.forEach(icon => {
                        icon.style.transform = isAnyClosed ? 'rotate(90deg)' : 'rotate(0deg)';
                    });

                    toggleAllBtn.innerHTML = isAnyClosed ?
                        '<i class="fas fa-compress-arrows-alt"></i> Zatvori sve foldere' :
                        '<i class="fas fa-expand-arrows-alt"></i> Otvori sve foldere';
                });
            }

            // Default keep all folders and sections CLOSED
            document.querySelectorAll('.folder-content, .section-content').forEach(content => {
                content.classList.remove('open');
                const header = content.previousElementSibling;
                if (header) {
                    const icon = header.querySelector('.folder-icon');
                    if (icon && header.classList.contains('folder-header')) {
                         icon.style.transform = 'rotate(0deg)';
                    }
                }

        // ==================== PERSISTENT STORAGE (ISPRAVLJENO) ====================
        function saveToStorage() {
            try {
                const data = {
                    structure: serializeStructure(),
                    timestamp: Date.now(),
                    version: CURRENT_VERSION
                };
                localStorage.setItem(STORAGE_KEY, JSON.stringify(data));
                console.log('Data saved to localStorage with version:', CURRENT_VERSION);
            } catch (e) {
                console.error('Failed to save to localStorage:', e);
                showStatus('Gre≈°ka pri ƒçuvanju podataka!', 'error');
            }
        }

        function restoreFromStorage() {
            try {
                const stored = localStorage.getItem(STORAGE_KEY);
                if (stored) {
                    const data = JSON.parse(stored);

                    if (!data.version || data.version !== CURRENT_VERSION) {
                        console.log('Version mismatch or missing version. Clearing old data.');
                        localStorage.removeItem(STORAGE_KEY);
                        return false;
                    }

                    if (data.structure && data.structure.sections) {
                        deserializeStructure(data.structure);
                        console.log('Data restored from localStorage');
                        return true;
                    }
                }
            } catch (e) {
                console.error('Failed to restore from localStorage:', e);
                localStorage.removeItem(STORAGE_KEY);
            }
            return false;
        }

        // Robustija funkcija za serijalizaciju koja izbegava direktnu manipulaciju `childNodes`.
        function serializeStructure() {
            const sections = [];
            document.querySelectorAll('.main-section').forEach((section) => {
                const header = section.querySelector('.section-header');
                if (!header) return;

                const headerClone = header.cloneNode(true);
                const icon = headerClone.querySelector('.folder-icon, .icon');
                const iconHTML = icon ? icon.outerHTML : '';
                
                if (icon) {
                    icon.remove(); // Uklanjamo ikonu iz klona da dobijemo ƒçist tekst
                }
                const headerText = headerClone.textContent.trim();
                
                const folders = [];
                serializeFolder(section.querySelector('.section-content'), folders, 1);

                sections.push({
                    id: section.dataset.id || `section_${sections.length + 1}`,
                    headerText: headerText,
                    iconHTML: iconHTML,
                    folders: folders
                });
            });
            return { sections, nextElementId };
        }

        function serializeFolder(container, folders, level) {
            if (!container) return;

            container.querySelectorAll(':scope > .folder').forEach((folder) => {
                const header = folder.querySelector('.folder-header .folder-name');
                const folderName = header ? header.textContent.trim() : '';
                const content = folder.querySelector('.folder-content');
                const bookmarks = [];

                if (content) {
                    content.querySelectorAll(':scope > .leaf').forEach(leaf => {
                        const link = leaf.querySelector('a');
                        if (link) {
                            bookmarks.push({
                                title: link.textContent.trim(),
                                url: link.href
                            });
                        }
                    });
                }

                const childFolders = [];
                if (content) {
                    serializeFolder(content, childFolders, level + 1);
                }

                folders.push({
                    id: folder.dataset.id || `folder_${folders.length + 1}`,
                    name: folderName,
                    level: level,
                    bookmarks: bookmarks,
                    children: childFolders,
                    isOpen: content ? content.classList.contains('open') : false
                });
            });
        }
        
        // Funkcija za deserijalizaciju ostaje ista jer je problem bio u podacima koje dobija.
        function deserializeStructure(structure) {
            const container = document.querySelector('.container .content');
            if (!container) {
                console.error("Container .content not found for deserialization");
                return;
            }

            container.innerHTML = ''; // Clear existing content

            structure.sections.forEach((sectionData) => {
                const section = document.createElement('div');
                section.className = 'main-section';
                section.dataset.id = sectionData.id;

                const iconHTML = sectionData.iconHTML || '';
                const headerText = sectionData.headerText || '';
                
                // Sklapamo header od saƒçuvanih delova
                section.innerHTML = `
                    <div class="section-header">
                        ${iconHTML}
                        ${headerText}
                    </div>
                    <div class="section-content"></div>
                `;

                const sectionContent = section.querySelector('.section-content');
                if (sectionData.folders) {
                    sectionData.folders.forEach(folderData => {
                        createFolderElement(sectionContent, folderData);
                    });
                }

                container.appendChild(section);
            });

            if (structure.nextElementId) {
                nextElementId = structure.nextElementId;
            }

            setTimeout(() => {
                attachFolderClickHandlers();
                cacheAllFolders();
            }, 100);
        }

        function createFolderElement(container, folderData) {
            const folderDiv = document.createElement('div');
            folderDiv.className = `folder level-${folderData.level}`;
            folderDiv.dataset.id = folderData.id;

            folderDiv.innerHTML = `
                <div class="folder-header">
                    <i class="fas fa-chevron-right folder-icon" style="transform: ${folderData.isOpen ? 'rotate(90deg)' : 'rotate(0deg)'}"></i>
                    <span class="folder-name">${folderData.name}</span>
                </div>
                <div class="folder-content ${folderData.isOpen ? 'open' : ''}"></div>
            `;
            const folderContent = folderDiv.querySelector('.folder-content');

            folderData.bookmarks.forEach(bookmarkData => {
                const leafDiv = document.createElement('div');
                leafDiv.className = 'leaf';
                leafDiv.innerHTML = `<a href="${bookmarkData.url}" target="_blank" rel="noopener noreferrer">${bookmarkData.title}</a>`;
                folderContent.appendChild(leafDiv);
            });

            folderData.children.forEach(childData => {
                createFolderElement(folderContent, childData);
            });

            container.appendChild(folderDiv);
        }

        function cacheAllFolders() {
            allFolders = Array.from(document.querySelectorAll('.folder, .main-section')).map((element, index) => {
                let name = '';
                let level = 0;
                let header = null;

                if (element.classList.contains('main-section')) {
                    header = element.querySelector('.section-header');
                    const icon = header.querySelector('.icon, .folder-icon');
                    let text = '';
                    header.childNodes.forEach(node => {
                        if (node !== icon) text += node.textContent;
                    });
                    name = text.trim();
                    level = 0;
                } else {
                    header = element.querySelector('.folder-header');
                    const folderNameEl = header ? header.querySelector('.folder-name') : null;
                    name = folderNameEl ? folderNameEl.textContent.trim() : '';
                    level = getFolderLevel(element);
                }

                return {
                    element: element,
                    header: header,
                    name: name,
                    level: level,
                    index: index,
                    id: element.dataset.id || `element_${index}`
                };
            });
        }

        function getFolderLevel(folder) {

        }

        function findBookmarkCategory(link) {
            let element = link.closest('.folder, .main-section');
            if (element) {
                const header = element.querySelector('.folder-name, .section-header');
                if (header) return header.textContent.trim();

            return 'Nepoznata kategorija';
        }

        function parseNumbering(text) {
            const match = text.match(/^(\d+(?:\.\d+)*)/);
            return match ? match[1].split('.').map(n => parseInt(n, 10)) : null;
        }

        function generateMainSectionNumber() {
            let maxNumber = 0;
            document.querySelectorAll('.main-section .section-header').forEach(header => {
                const numbers = parseNumbering(header.textContent.trim());
                if (numbers && numbers.length > 0) {
                    maxNumber = Math.max(maxNumber, numbers[0]);
                }
            });
            return maxNumber + 1;
        }

        function generateFolderNumber(parentFolderId) {
            const parent = allFolders.find(f => f.id === parentFolderId);
            if (!parent) return '1';

            const parentNumbers = parseNumbering(parent.name) || [];
            const parentContent = parent.element.querySelector('.folder-content, .section-content');
            if (!parentContent) return parentNumbers.join('.') + '.1';

            let maxChild = 0;
            parentContent.querySelectorAll(':scope > .folder').forEach(childFolder => {
                const childHeader = childFolder.querySelector('.folder-header .folder-name');
                if (childHeader) {
                    const childNumbers = parseNumbering(childHeader.textContent);
                    if (childNumbers && childNumbers.length === parentNumbers.length + 1) {
                        maxChild = Math.max(maxChild, childNumbers[childNumbers.length - 1]);
                    }
                }
            });

            return [...parentNumbers, maxChild + 1].join('.');
        }

        function insertMainSectionAt(newSection, position, referenceSectionId = null) {
            const container = document.querySelector('.container .content');
            if (!container) {
                console.error("Container .content not found!");
                return;
            }

            const refSection = referenceSectionId ? document.querySelector(`.main-section[data-id="${referenceSectionId}"]`) : null;

            switch (position) {
                case 'root-start':
                    container.prepend(newSection);
                    break;
                case 'root-end':
                    container.appendChild(newSection);
                    break;
                case 'root-before':
                    if (refSection) refSection.before(newSection);
                    else container.prepend(newSection); // Fallback
                    break;
                case 'root-after':
                    if (refSection) refSection.after(newSection);
                    else container.appendChild(newSection); // Fallback
                    break;
                default:
                    container.appendChild(newSection);
            }
            reindexMainSections();
        }

        // Robustija funkcija za reindeksiranje koja izbegava direktnu manipulaciju `childNodes`.
        function reindexMainSections() {
            const sections = document.querySelectorAll('.main-section');
            sections.forEach((section, index) => {
                const header = section.querySelector('.section-header');
                if (header) {
                    const originalIcon = header.querySelector('.folder-icon, .icon');
                    const headerClone = header.cloneNode(true);
                    const iconInClone = headerClone.querySelector('.folder-icon, .icon');
                    
                    if (iconInClone) {
                        iconInClone.remove();
                    }
                    
                    const textContent = headerClone.textContent.trim();
                    const currentText = textContent.replace(/^\d+\.\s*/, '');
                    const newNumber = index + 1;
                    
                    header.innerHTML = ''; // Bri≈°emo stari sadr≈æaj
                    if (originalIcon) {
                        header.appendChild(originalIcon.cloneNode(true)); // Vraƒáamo originalnu ikonu
                    }
                    // Dodajemo razmak, novi broj i ƒçist tekst kao text node
                    header.append(` ${newNumber}. ${currentText}`);
                }
            });
        }

        function addBookmark() {
            const title = document.getElementById('bookmarkTitle').value.trim();
            let url = document.getElementById('bookmarkUrl').value.trim();
            const category = document.getElementById('categorySelect').value;

            if (!title || !url || !category) {
                showStatus('Molimo popunite sva polja!', 'error');
                return;
            }
            if (!url.match(/^https?:\/\//)) {
                url = 'https://' + url;
            }

            let targetElement = null;
            document.querySelectorAll('.folder-name, .section-header').forEach(el => {
                if (el.textContent.trim() === category) {
                    targetElement = el.closest('.folder, .main-section').querySelector('.folder-content, .section-content');
                }
            });

            if (!targetElement) {
                showStatus('Kategorija nije pronaƒëena!', 'error');
                return;
            }

            const newBookmark = document.createElement('div');
            newBookmark.className = 'leaf';
            newBookmark.innerHTML = `<a href="${url}" target="_blank" rel="noopener noreferrer">${title}</a>`;
            targetElement.appendChild(newBookmark);
            setBlank([newBookmark.querySelector('a')]);

            saveToStorage();
            cacheAllFolders();
            showStatus('Bookmark je uspe≈°no dodat!', 'success');
            setTimeout(closeBookmarkModal, 1500);
        }

        function addFolder() {
            const folderName = document.getElementById('folderName').value.trim();
            const parentFolderId = document.getElementById('parentFolderSelect').value;

            if (!folderName) {
                showStatus('Molimo unesite naziv foldera!', 'error');
                return;
            }

            const selectedPosition = document.querySelector('input[name="position"]:checked').value;

            if (!parentFolderId || parentFolderId === '') { // Adding a new Main Section
                const newSection = document.createElement('div');
                newSection.className = 'main-section';
                newSection.dataset.id = `section_${nextElementId++}`;
                newSection.innerHTML = `
                <div class="section-header">
                    <i class="fas fa-folder icon"></i>
                    ${folderName}
                </div>
                <div class="section-content"></div>
            `;
                let refId = null;
                if (selectedPosition === 'root-before' || selectedPosition === 'root-after') {
                    const sectionSelect = document.getElementById('sectionSelect');
                    if (!sectionSelect || !sectionSelect.value) {
                        showStatus('Molimo izaberite referentnu sekciju!', 'error');
                        return;
                    }
                    refId = sectionSelect.value;
                }
                insertMainSectionAt(newSection, selectedPosition, refId);
            } else { // Adding a sub-folder
                const parent = allFolders.find(f => f.id === parentFolderId);
                if (!parent) {
                    showStatus('Parent folder nije pronaƒëen!', 'error');
                    return;
                }
                const targetLevel = parent.level + 1;
                if (targetLevel > 5) { // CSS podr≈æava do 4, ali logika dozvoljava vi≈°e
                    showStatus('Maksimalna dubina je 5 nivoa!', 'error');
                    return;
                }
                const folderNumber = generateFolderNumber(parentFolderId);
                const newFolder = document.createElement('div');
                newFolder.className = `folder level-${targetLevel}`;
                newFolder.dataset.id = `folder_${nextElementId++}`;
                newFolder.innerHTML = `
                <div class="folder-header">
                    <i class="fas fa-chevron-right folder-icon"></i>
                    <span class="folder-name">${folderNumber} ${folderName}</span>
                </div>
                <div class="folder-content"></div>
            `;
                if (selectedPosition === 'child') {
                    const container = parent.element.querySelector('.folder-content, .section-content');
                    if (container) {
                        container.appendChild(newFolder);
                        container.classList.add('open');
                        const icon = parent.header.querySelector('.folder-icon');
                        if (icon) icon.style.transform = 'rotate(90deg)';
                    }
                } else if (selectedPosition === 'sibling-before') {
                    parent.element.before(newFolder);
                } else if (selectedPosition === 'sibling-after') {
                    parent.element.after(newFolder);
                }
            }
            saveToStorage();
            attachFolderClickHandlers();
            cacheAllFolders();
            showStatus('Folder je uspe≈°no dodat!', 'success');
            setTimeout(closeBookmarkModal, 1500);
        }

        function removeBookmark() {
            const selectValue = document.getElementById('bookmarkRemoveSelect').value;
            if (!selectValue) {
                showStatus('Molimo izaberite bookmark za brisanje!', 'error');
                return;
            }
            const bookmark = bookmarkElements[parseInt(selectValue)];
            if (bookmark && bookmark.element) {
                bookmark.element.remove();
                saveToStorage();
                cacheAllFolders();
                showStatus('Bookmark je uspe≈°no uklonjen!', 'success');
                setTimeout(closeBookmarkModal, 1500);
            } else {
                showStatus('Gre≈°ka pri brisanju bookmark-a!', 'error');
            }
        }

        function removeFolder() {
            const selectValue = document.getElementById('folderRemoveSelect').value;
            const deleteOption = document.getElementById('deleteOptionsSelect').value;
            if (!selectValue) {
                showStatus('Molimo izaberite folder za brisanje!', 'error');
                return;
            }
            const folder = allFolders.find(f => f.id === selectValue);
            if (!folder) {
                showStatus('Folder nije pronaƒëen!', 'error');
                return;
            }
            if (deleteOption === 'folder-only') {
                const parent = folder.element.parentElement;
                const content = folder.element.querySelector('.folder-content, .section-content');
                if (content && parent) {
                    while (content.firstChild) {
                        parent.insertBefore(content.firstChild, folder.element);
                    }
                }
            }
            const isMainSection = folder.element.classList.contains('main-section');
            folder.element.remove();
            if (isMainSection) {
                reindexMainSections();
            }
            saveToStorage();
            cacheAllFolders();
            showStatus('Folder je uspe≈°no uklonjen!', 'success');
            setTimeout(closeBookmarkModal, 1500);
        }

        function showStatus(message, type) {
            const statusDiv = document.getElementById('statusMessage');
            statusDiv.textContent = message;
            statusDiv.className = `status-message status-${type}`;
        }

        // ==================== EXPORT HTML FUNCTIONALITY ====================
        function exportHTML() {
            try {
                const docClone = document.documentElement.cloneNode(true);
                docClone.querySelectorAll('[data-click-attached], [data-sb-highlight], [data-sb-hidden]').forEach(el => {
                    el.removeAttribute('data-click-attached');
                    el.removeAttribute('data-sb-highlight');
                    el.removeAttribute('data-sb-hidden');
                    if (el._sb_prev) delete el._sb_prev;
                });
                // =========== ISPRAVKA ===========
                // Sledeƒáa linija je komentarisana da bi se saƒçuvali svi interaktivni elementi
                // docClone.querySelectorAll('.floating-button, .modal, #searchBar').forEach(el => el.remove());
                // ================================
                const htmlString = '<!DOCTYPE html>\n' + docClone.outerHTML;
                const blob = new Blob([htmlString], {
                    type: 'text/html;charset=utf-8'
                });
                const url = URL.createObjectURL(blob);
                const link = document.createElement('a');
                link.href = url;
                link.download = 'lions-pith-database.html';
                document.body.appendChild(link);
                link.click();
                document.body.removeChild(link);
                URL.revokeObjectURL(url);
                showStatus('HTML je uspe≈°no izvezen!', 'success');
            } catch (error) {
                console.error('Export failed:', error);
                showStatus('Gre≈°ka pri izvozu HTML-a!', 'error');
            }
        }

        // ==================== SEARCH FUNCTIONALITY - ENTER MODE ====================
        let currentMatches = [];
        let currentMatchIndex = 0;

        function handleSearchKeydown(event) {
            if (event.key === 'Enter') {
                event.preventDefault();
                searchFolders();
            } else if (event.key === 'Escape') {
                clearSearchBar();
            }
        }

        function searchFolders() {
            const query = document.getElementById('searchBox').value.toLowerCase().trim();
            resetAllSearchEffects();

            if (!query) return;

            const candidates = Array.from(document.querySelectorAll('a, .folder-header, .section-header'));
            currentMatches = candidates.filter(n => (n.textContent || '').toLowerCase().includes(query));

            updateResultsDisplay();

            if (currentMatches.length > 0) {
                const elementsToShow = new Set();
                currentMatches.forEach(match => {
                    let el = match;
                    while (el) {
                        elementsToShow.add(el);
                        el = el.parentElement;
                    }
                });

                document.querySelectorAll('.main-section, .folder').forEach(el => {
                    if (!elementsToShow.has(el)) {
                        el.setAttribute('data-sb-hidden', 'true');
                        el.style.display = 'none'; // Hide non-matching elements
                    }
                });
                 
                elementsToShow.forEach(el => {
                    if (el.style.display === 'none') {
                        el.style.display = '';
                    }
                    if (el.classList.contains('folder') || el.classList.contains('main-section')) {
                        const content = el.querySelector(':scope > .folder-content, :scope > .section-content');
                        if (content) content.classList.add('open');
                        const icon = el.querySelector(':scope > .folder-header .folder-icon, :scope > .section-header .folder-icon');
                        if (icon) icon.style.transform = 'rotate(90deg)';
                    }
                });

                highlightResult(0);
            }
        }

        function resetAllSearchEffects() {
            document.querySelectorAll('[data-sb-highlight="true"]').forEach(el => {
                el.removeAttribute('data-sb-highlight');
                if (el._sb_prev) {
                    el.style.outline = el._sb_prev.outline || '';
                    el.style.backgroundColor = el._sb_prev.backgroundColor || '';
                    delete el._sb_prev;
                }
            });
            document.querySelectorAll('[data-sb-hidden]').forEach(el => {
                 el.removeAttribute('data-sb-hidden');
                 el.style.display = '';
            });
            currentMatches = [];
            currentMatchIndex = 0;
            updateResultsDisplay();
        }

        function updateResultsDisplay() {
            const resultsDiv = document.getElementById('searchResults');
            const navigationDiv = document.getElementById('searchNavigation');
            const query = document.getElementById('searchBox').value;
            if (currentMatches.length === 0) {
                resultsDiv.textContent = query ? 'Nema rezultata' : '';
                navigationDiv.style.display = 'none';
                return;
            }
            resultsDiv.textContent = `Pronaƒëeno: ${currentMatches.length} rezultata`;
            if (currentMatches.length > 1) {
                navigationDiv.style.display = 'block';
                document.getElementById('currentResult').textContent = `${currentMatchIndex + 1}/${currentMatches.length}`;
                document.getElementById('prevResult').disabled = currentMatchIndex === 0;
                document.getElementById('nextResult').disabled = currentMatchIndex === currentMatches.length - 1;
            } else {
                navigationDiv.style.display = 'none';
            }
        }

        function highlightResult(index) {
            document.querySelectorAll('[data-sb-highlight="true"]').forEach(el => {
                el.removeAttribute('data-sb-highlight');
                if (el._sb_prev) {
                    el.style.outline = el._sb_prev.outline;
                    el.style.backgroundColor = el._sb_prev.backgroundColor;
                    delete el._sb_prev;
                }
            });
            if (index < 0 || index >= currentMatches.length) return;
            const target = currentMatches[index];
            target.scrollIntoView({
                behavior: 'smooth',
                block: 'center'
            });
            target._sb_prev = {
                outline: target.style.outline || '',
                backgroundColor: target.style.backgroundColor || ''
            };
            target.setAttribute('data-sb-highlight', 'true');
            target.style.outline = '2px solid #4a9eff';
            target.style.backgroundColor = '#333844';
        }

        function navigateResults(direction) {
            currentMatchIndex += direction;
            if (currentMatchIndex < 0) currentMatchIndex = 0;
            if (currentMatchIndex >= currentMatches.length) currentMatchIndex = currentMatches.length - 1;
            updateResultsDisplay();
            highlightResult(currentMatchIndex);
        }

        function clearSearchBar() {
            const searchBox = document.getElementById('searchBox');
            searchBox.value = '';
            resetAllSearchEffects();
            searchBox.focus();
        }

        // ==================== LINK HELPER ====================
        function setBlank(links) {
            links.forEach(a => {
                a.setAttribute('target', '_blank');
                a.setAttribute('rel', 'noopener noreferrer');
            });
        }

        new MutationObserver(mutations => {
            mutations.forEach(m => {
                m.addedNodes.forEach(node => {
                    if (node.nodeType !== 1) return;
                    const links = node.matches('a') ? [node] : node.querySelectorAll('a');
                    setBlank(links);
                });
            });
        }).observe(document.body, {
            childList: true,
            subtree: true
        });
    </script>
